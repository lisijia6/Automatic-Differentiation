4def Newton(f: callable, *x0, tol=1e-5, max_iter=500, n_iter=1): 

5 r""" 

6 Newton's method 

7 

8 To find :math:`\mathbf{x}` such that :math:`F(\mathbf{x}) = \mathbf{0}`, we need update :math:`\Delta \mathbf{x}_{k}` such that 

9 

10 .. math:: 

11 J_F(\mathbf{x}_k)\Delta \mathbf{x}_{k} &= - F(\mathbf{x}_{k}) \\ 

12 \mathbf{x}_{k+1} &\gets \mathbf{x}_{k} + \Delta \mathbf{x}_{k} 

13 where :math:`J_F(\mathbf{x}_k)` is the Jacobian and :math:`\Delta \mathbf{x}_{k}` is the update. 

14  

15 :param f: A callable function object, the :math:`F: \mathbb{R}^m \mapsto \mathbb{R}^n` function 

16 :type f: function object 

17 :param x0: The initial guess, note that Newton is quadratic convergence if initial guess is close to the actual solution 

18 :type x0: integer or float or numpy array or list of integers or floats 

19 :param tol: The tolerance, the algorithm terminates when it hits the tolerance i.e. when :math:`\|F(\mathbf{x})\|_F < \text{tol}` is reached, the algorithm terminates 

20 :type tol: float 

21 :raises RuntimeError: If the function does not converge in max_iter iterations 

22  

23 :return: The solution :math:`\mathbf{x}` 

24 :rtype: float or list of floats 

25 

26 >>> x0 = [0, 1] 

27 >>> def f(x1, x2): 

28 >>> return [ 

29 >>> 2 * x1 + x2 - np.exp(-x1), 

30 >>> -x1 + 2 * x2 - np.exp(-x2) 

31 >>> ] 

32 >>> sol = Newton(f, *x0) 

33 >>> sol 

34 array([0.19759433, 0.42551406])  

35 >>> f(*sol) 

36 [-4.3786574366322384e-10, -3.137059279012533e-09] 

37 """ 

38 n_iter += 1 

39 if n_iter == max_iter: 

40 raise RuntimeError(f'The function does not converge in {n_iter} iterations!') 

41 x0 = np.array(x0) 

42 # if the norm of the function is less than the tolerance, consider the method converged 

43 if np.linalg.norm(f(*x0)) < tol: 

44 # if result list has length 1, return the number without the bracket 

45 if len(x0) == 1: 

46 return x0.item() 

47 return x0 

48 # use Forward AD for derivative calculation 

49 g = Forward(f, *x0) 

50 if isinstance(g.der, (int, float)): 

51 # if the derivative g is a number, perform Newton's Method in 1D 

52 new_x = x0 - g.val / g.der 

53 else: 

54 # else perform Newton's Method in nD 

55 if g.der.ndim == 1: 

56 g.der = g.der.reshape(1, -1) 

57 g.val = np.array(g.val).reshape(1) 

58 update, *_ = np.linalg.lstsq(g.der, -g.val, rcond=None) 

59 new_x = x0 + update 

60 return Newton(f, *new_x, tol=tol, n_iter=n_iter)
